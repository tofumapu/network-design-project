## Password cho Router khi login qua cong console & telnet
conf t

password cisco 
login
exit

password cisco
transport input telnet

## Password khi vao che do privileged mode (khi enable)
(config)#enable password cisco
(config)#enable secret cisco
## Ma hoa password
(config)#service password-encryption
(config)# no service password-encryption (neu muon tat)

#Dat ten router
(config)# hostname HOSTNAME

#Dat banner cho router
(config)# banner motd #FastPay HCM xin chao#

#Dat dia chi IP cho interface
Router(config)#interface fastethernet0/0
Router(config-if)#ip address 192.168.1.1 255.255.255.0 
Router(config-if)#no shutdown 
##Dat cho cong serial
Router(config)# interface  serial 0/0  
//Gọi tên cổng cần cấu hình 
Router(config-if)#ip address 192.168.1.1 255.255.255.0 
Router(config-if)#clock rate 64000  # Giua cac router xung nhip dong bo voi nhau 
Router(config-if)#no shutdown

# Description cho interface
Router(config)#interface fa0/0 
Router(config-if)#description Kết nối với LAN Sales

# Cau lenh config
Router#show ip route: Xem bảng định tuyến 
Router#Show version: Xem cấu hình thanh ghi 
Router#Show flash: Xem thông tin hệ điều hành (.bin) 
Router#Show history: Xem tất cả lệnh trong bộ nhớ(buffer) 
Router#Show ip interface brief 
Router#Show cdp: Xem thông tin CDP 
Router#ping : Kiễm tra kết nối 
Router#Show running-config // hiển thị cấu hình in RAM 
Router#Show start-config //hiển thị cấu hình in NVRAM   

#Luu tru file cau hinh
Router#copy running-config startup-config
